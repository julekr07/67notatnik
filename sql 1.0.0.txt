CREATE DATABASE school_api CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE school_api;

-- Użytkownicy
CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  isTeacher BOOLEAN NOT NULL DEFAULT FALSE,
  password VARCHAR(255) NOT NULL
);

-- Wiadomości
CREATE TABLE messages (
  id INT AUTO_INCREMENT PRIMARY KEY,
  userId INT NOT NULL,
  content TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (userId) REFERENCES users(id)
);

-- Notatki prywatne
CREATE TABLE notes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  userId INT NOT NULL,
  content TEXT NOT NULL,
  FOREIGN KEY (userId) REFERENCES users(id)
);

-- Tablica nauczyciela (jedna globalna)
CREATE TABLE board (
  id INT AUTO_INCREMENT PRIMARY KEY,
  content TEXT NOT NULL
);
